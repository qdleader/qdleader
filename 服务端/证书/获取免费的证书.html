йҖүжӢ©--гҖӢ иҙӯд№°иҜҒд№Ұ

--гҖӢ е…Қиҙ№зүҲжң¬

д№°е®ҢжӯЈеёёй…ҚзҪ®еҚіеҸҜгҖӮ

жңҖеҗҺ иҜҒд№ҰдёӢиҪҪдёӢжқҘпјҢеңЁжңҚеҠЎеҷЁй…ҚзҪ®ж—¶еҖҷ

е®ҳж–№зҡ„nginx еҶҷжі•еҸҲй—®йўҳпјҢдјҡжҠҘй”ҷ

server {
listen 443;
server_name localhost; # localhost дҝ®ж”№дёәжӮЁиҜҒд№Ұз»‘е®ҡзҡ„еҹҹеҗҚгҖӮ
ssl on; #и®ҫзҪ®дёә on еҗҜз”Ё SSL еҠҹиғҪгҖӮ
root html;
index index.html index.htm;
ssl_certificate cert/domain name.pem; #е°Ҷ domain name.pem жӣҝжҚўжҲҗжӮЁиҜҒд№Ұзҡ„ж–Үд»¶еҗҚгҖӮ
ssl_certificate_key cert/domain name.key; #е°Ҷ domain name.key жӣҝжҚўжҲҗжӮЁиҜҒд№Ұзҡ„еҜҶй’Ҙж–Үд»¶еҗҚ
 ssl_session_timeout 5m;
ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE:ECDH:AES:HIGH:!NULL:!aNULL:!MD5:!ADH:!RC4; #дҪҝз”ЁжӯӨеҠ еҜҶеҘ—д»¶гҖӮ
ssl_protocols TLSv1 TLSv1.1 TLSv1.2; #дҪҝз”ЁиҜҘеҚҸи®®иҝӣиЎҢй…ҚзҪ®гҖӮ
 ssl_prefer_server_ciphers on;
location / {
root html; #з«ҷзӮ№зӣ®еҪ•гҖӮ
index index.html index.htm; #ж·»еҠ еұһжҖ§гҖӮ
}
