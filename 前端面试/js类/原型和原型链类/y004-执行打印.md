# y004-执行打印
```js
let aname = '2'
obj = {
  aname : '1',
  say() {
    console.log(this.aname);
    return function () {
      console.log(this.aname)
      console.log(aname)
    }
  }
}
obj.say()()

```
```js
1
undefined
2

```
