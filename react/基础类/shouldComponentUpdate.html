shouldComponentUpdate

еҸӘиҰҒжҳҜз»„д»¶з»§жүҝиҮӘReact.Componentе°ұдјҡеӯҳеңЁеҪ“зҲ¶з»„д»¶йҮҚж–°renderзҡ„ж—¶еҖҷпјҢеӯҗз»„д»¶д№ҹдјҡйҮҚж–°жү§иЎҢrender,еҚідҪҝиҝҷдёӘеӯҗз»„д»¶жІЎжңүд»»дҪ•еҸҳеҢ–гҖӮеӯҗз»„д»¶еҸӘиҰҒи°ғз”ЁsetStateе°ұдјҡйҮҚж–°жү§иЎҢrender,еҸҠж—¶setStateзҡ„еҸӮж•°жҳҜдёҖдёӘз©әеҜ№иұЎгҖӮ


shouldComponentUpdate(nextProps,nextState) {
    console.log(this.props,this.state); // зӣ®еүҚзҡ„props е’Ң state
    console.log(nextProps,nextState); // иҰҒеҸҳеҢ–зҡ„зӣ®ж Ү props зҡ„е’Ң зӣ®ж Ү state
    if (nextProps.m1 === this.props.m1 && nextState.m2 === this.state.m2) {
        return false;   // жІЎжңүеҸҳпјҢ дёҚжёІжҹ“дәҶ
    } else {
        return true;  // ж”№еҸҳдәҶ жүҚжёІжҹ“
    }
}

еҪ“shouldComponentUpdateиҝ”еӣһдёәtrueзҡ„ж—¶еҖҷпјҢеҪ“еүҚз»„д»¶иҝӣиЎҢrenderпјҢеҰӮжһңиҝ”еӣһзҡ„жҳҜfalseеҲҷдёҚиҝӣиЎҢrender.
