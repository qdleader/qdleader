var isValid = function(s) {
  if(s % 2 == 1 ) {
    return false
  }
  let arr1 = [];
  for(let i = 0;i < s.length; i ++) {
    if(s[i] == '(' || s[i] == '[' || s[i] == '{') {
      arr1.push(s[i])
    }
  }
  if(arr1.length != s.length/2) {
    return false
  }
  let arr = [];
  for(let i = 0;i < s.length; i ++) {
    if(s[i] == '(' || s[i] == '[' || s[i] == '{') {
      arr.push(s[i])
    } else {
      let item = arr[arr.length - 1]
      if(item == '(' && s[i] != ')' ) {
        return false
      } else if(item == '[' && s[i] != ']') {
        return false
      } else if(item == '{' && s[i] != '}') {
        return false
      }
      arr.pop()
    }
  }
  return arr.length == 0
};
