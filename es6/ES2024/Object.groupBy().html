Object.groupBy()

const students = [
  { name: "Alice", grade: "A" },
  { name: "Bob", grade: "B" },
  { name: "Charlie", grade: "A" },
  { name: "David", grade: "C" }
];

const studentsByGrade = Object.groupBy(students, student => student.grade);
console.log(studentsByGrade);
