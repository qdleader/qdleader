## CSRпјҲClient Side Renderingпјү

CSR д№ҹе°ұжҳҜе®ўжҲ·з«ҜжёІжҹ“пјҢйңҖиҰҒдҪҝз”Ё JavaScriptпјҢи°ғз”ЁжҺҘеҸЈпјҲAPIпјүжқҘиҺ·еҸ–ж•°жҚ®пјҢиҝҷз§Қж–№ејҸеүҚеҗҺз«Ҝе®Ңе…ЁеҲҶзҰ»гҖӮ
жҜ”еҰӮзҺ°еңЁжңүдёҖдёӘеҚҡе®ўжҺҘеҸЈ/api/articlesпјҢиҝ”еӣһ JSON ж•°жҚ®еҰӮдёӢ

js
[
    {"id":1,"title":"дҪҝз”Ё Next.js е’ҢжҺҳйҮ‘API жү“йҖ дёӘжҖ§еҚҡе®ў"},
    {"id":2,"title":"дҪҝз”Ё Strapi е’Ң Next.js ејҖеҸ‘з®Җжҳ“еҫ®еҚҡ"},
    {"id":3,"title":"дҪҝз”Ё Notion ж•°жҚ®еә“иҝӣиЎҢ Next.js еә”з”Ёе…Ёж ҲејҖеҸ‘"}
]

йҖҡеёё React йЎ№зӣ®дјҡдҪҝз”Ё create-react-app жқҘеҲӣе»әйЎ№зӣ®пјҢжҲ‘д»¬дјҡеңЁuseEffect дёӯиҜ·жұӮж•°жҚ®гҖӮ
js
import { useState, useEffect } from "react";

function BlogList() {
  const [data, setData] = useState(null);
  const [isLoading, setLoading] = useState(false);

  const reload = () => {
    setLoading(true);
    fetch("/api/articles").then((res) => res.json()).then((data) => {
        setData(data);
        setLoading(false);
      });
  };

  useEffect(() => {
    reload();
  }, []);

  if (!data && isLoading) return <p>Loading...</p>;
  if (!data) return <p>No data</p>;

  return (
    <div>
      <div>
        <button disabled={isLoading} loading={isLoading} onClick={reload}>еҲ·ж–°</button>
      </div>
      {data.map((item) => (
        <div key={item.id}>
          <a href={`/blog/${item.id}`}>{item.title}</a>
        </div>
      ))}
    </div>
  );
}

дёҠйқўзҡ„д»Јз ҒдёӯпјҢйЎөйқўдёҠиҝҳжңүдёҖдёӘеҲ·ж–°жҢүй’®пјҢеҪ“ж•°жҚ®ж–°еўһж—¶пјҢжҺҘеҸЈжҺҘеҸЈдјҡеӨҡиҝ”еӣһдёҖжқЎж•°жҚ®пјҢзӮ№еҮ»еҲ·ж–°жҢүй’®пјҢйЎөйқўдёҠе·Із»ҸеӯҳеңЁзҡ„ DOM иҠӮзӮ№жҳҜдёҚжӣҙж–°зҡ„пјҢDOM дёӯеҸӘдјҡжҸ’е…Ҙж–°еўһзҡ„ж•°жҚ®пјҢиҝҷж ·жҲ‘д»¬е°ұдјҡж„ҹи§үйЎөйқўжёІжҹ“еҫҲеҝ«гҖӮ
иҝҷеҫ—зӣҠдәҺ React дёӯеј•е…ҘдәҶиҷҡжӢҹ domпјҢд№ҹе°ұжҳҜе°Ҷзңҹе®һе…ғзҙ иҠӮзӮ№жҠҪиұЎжҲҗ JavaScript еҜ№иұЎпјҢз§°д№Ӣдёә VNodeпјҢжӣҙж–° DOM еүҚдјҡе…ҲйҖҡиҝҮ VDOM еҜ№жҜ”пјҢеҫ—еҲ°иҰҒзңҹе®һжӣҙж–°зҡ„ DOMпјҢеӣ жӯӨеҸҜд»Ҙжңүж•ҲеҮҸе°‘зӣҙжҺҘж“ҚдҪң dom ж¬Ўж•°пјҢд»ҺиҖҢжҸҗй«ҳзЁӢеәҸжҖ§иғҪгҖӮ

Next.js еӣўйҳҹеҸ‘еёғдәҶеҸҰдёҖдёӘе…ідәҺж•°жҚ®иҜ·жұӮзҡ„ hooks еҸ« swrпјҢеҗҚеӯ—жқҘиҮӘдәҺ stale-while-revalidateпјҢж„ҸжҖқжҳҜиҝҮжңҹе°ұдјҡйҮҚж–°йӘҢиҜҒпјҢе®ғжңүзј“еӯҳпјҢиҒҡз„Ұж—¶йҮҚж–°йӘҢиҜҒпјҢй—ҙйҡ”иҪ®иҜўзӯүеҠҹиғҪгҖӮ
дёҺдёҠйқўд»Јз ҒеҠҹиғҪзӣёеҗҢпјҢжҲ‘д»¬еҸҜд»Ҙж”№жҲҗдёӢйқўд»Јз Ғпјҡ
js
import useSWR from "swr";

const fetcher = (...args) => fetch(...args).then((res) => res.json());

function BlogList() {
  const { data, error } = useSWR("/api/articles", fetcher, { refreshInterval: 3000 });

  if (error) return <div>Failed to load</div>;
  if (!data) return <div>Loading...</div>;

  return (
    <div>
      {data.map((item) => (
        <div key={item.id}>
          <a href={`/blog/${item.id}`}>{item.title}</a>
        </div>
      ))}
    </div>
  );
}

еҜ№дәҺжҗңзҙўеј•ж“ҺжқҘиҜҙпјҢеҸӘиғҪеңЁйЎөйқўдёӯеҸ‘зҺ°дёҖдёӘ DOM иҠӮзӮ№пјҢдёҚеҲ©дәҺ SEOпјӣеӣ дёәжҗңзҙўеј•ж“ҺжҳҜдёҚж”ҜжҢҒжү§иЎҢ JavaScript д»Јз Ғзҡ„гҖӮ

## SSRпјҲServer Side Renderingпјү

SSR д№ҹе°ұжҳҜжңҚеҠЎз«ҜжёІжҹ“пјҢжңүдәӣеҗҢеӯҰеҸҜиғҪдјҡй—®вҖңйҡҫйҒ“иҰҒеӣһеҲ° PHP жҲ–иҖ… JSP ж—¶д»Јеҗ—пјҹвҖқпјҢжІЎй”ҷ PHP е’Ң JSP жҳҜжңҚеҠЎз«ҜжёІжҹ“пјҢдҪҶ Next.js зҡ„ SSR дёҚеҗҢдәҺзәҜжңҚеҠЎз«ҜжёІжҹ“пјҢд№ҹжӢҘжңүзқҖеҰӮ SPA дёҖж ·еҝ«йҖҹжёІжҹ“зҡ„иғҪеҠӣгҖӮдј з»ҹзҡ„жңҚеҠЎз«ҜжёІжҹ“еҸӘжңү HTML еӯ—з¬ҰдёІпјҢзјәе°‘дәӨдә’пјҢжҜ”еҰӮжңүдёҖдёӘClickCounter з»„д»¶
js
// shared/components/ClickCounter.jsx
import React,{ useState } from 'react';

const ClickCounter = () => {
  const [count, setCount] = useState(0);
  return (
      <button onClick={() => setCount(count + 1)} > {count} Clicks </button>
  );
};

з»ҸиҝҮжңҚеҠЎз«ҜжёІжҹ“еҸӘиғҪеҫ—еҲ°жңҖз®ҖеҚ•зҡ„зҡ„ HTMLгҖӮ

js
// server/index.js
import ReactDOMServer from "react-dom/server";
import express from "express";

require("node-jsx").install();

const app = express();

app.get("/", (req, res) => {
    const reactHtml = ReactDOMServer.renderToString(<ClickCounter />пјү

    <html>
        <head>
            <title>Universal React server bundle</title>
        </head>
        <body>
            <div id="root">${reactHtml}</div>

        </body>

    res.send(htmlTemplate)

жү“еҚ°еҮәзҡ„ button зӮ№еҮ»ж— ж•ҲпјҢдј з»ҹзҡ„жңҚеҠЎз«ҜжёІжҹ“еҲ°жӯӨе°ұз»“жқҹдәҶгҖӮиҖҢ react жңҚеҠЎз«ҜжёІжҹ“пјҢйңҖиҰҒе®ўжҲ·з«Ҝж №жҚ®жңҚеҠЎз«Ҝз”ҹжҲҗзҡ„йЎөйқўпјҢз»§з»ӯдәҢж¬ЎжёІжҹ“гҖҒдәӢд»¶з»‘е®ҡзӯү

е®ўжҲ·з«Ҝж №жҚ®жёІжҹ“еҮәзҡ„йқҷжҖҒ html иҝӣиЎҢhydrateпјҢеҒҡдёҖдәӣз»‘е®ҡдәӢд»¶зӯүж“ҚдҪңгҖӮ
еӣ жӯӨпјҢиӢҘиҰҒдҪҝз”Ё react жқҘе®һзҺ°жңҚеҠЎз«ҜжёІжҹ“пјҢдёҖиҲ¬йңҖиҰҒ 3 дёӘзӣ®еҪ•пјҢе·ҘзЁӢй…ҚзҪ®жҜ”иҫғз№ҒзҗҗгҖӮ

#### SSR еӯҳеңЁзҡ„й—®йўҳ

> 1. еҪ“иҜ·жұӮйҮҸеўһеӨ§ж—¶пјҢжҜҸж¬ЎйҮҚж–°жёІжҹ“еўһеҠ дәҶжңҚеҠЎеҷЁзҡ„ејҖй”ҖгҖӮ
> 2. йңҖиҰҒзӯүйЎөйқўдёӯжүҖжңүжҺҘеҸЈиҜ·жұӮе®ҢжҲҗжүҚеҸҜд»Ҙиҝ”еӣһ htmlпјҢиҷҪдёҚжҳҜзҷҪеұҸпјҢдҪҶе®ҢжҲҗhydrateд№ӢеүҚпјҢйЎөйқўд№ҹжҳҜдёҚеҸҜж“ҚдҪңгҖӮ

## SSGпјҲStatic Site Generationпјү

SSG д№ҹе°ұжҳҜйқҷжҖҒз«ҷзӮ№з”ҹжҲҗпјҢдёәдәҶеҮҸзј“жңҚеҠЎеҷЁеҺӢеҠӣпјҢжҲ‘д»¬еҸҜд»ҘеңЁжһ„е»әж—¶з”ҹжҲҗйқҷжҖҒйЎөйқўпјҢеӨҮжіЁпјҡNext.js з”ҹжҲҗзҡ„йқҷжҖҒйЎөйқўдёҺжҷ®йҖҡзҡ„йқҷжҖҒйЎөйқўжҳҜдёҚдёҖж ·зҡ„пјҢд№ҹжҳҜжӢҘжңү SPA зҡ„иғҪеҠӣпјҢеҲҮжҚўйЎөйқўз”ЁжҲ·дёҚдјҡж„ҹеҸ—еҲ°ж•ҙдёӘйЎөйқўеңЁеҲ·ж–°
жҜ”еҰӮж–Үз« еҲ—иЎЁйЎөпјҢиҰҒз”ҹжҲҗйқҷжҖҒйЎөйқўпјҢеңЁ Next.js дёӯд»Јз ҒеҰӮдёӢпјҡ
js
import Link from "next/link";

дҪҝз”ЁgetStaticProps еҸҜд»ҘиҺ·еҫ—йқҷжҖҒзҪ‘йЎөзҡ„ж•°жҚ®пјҢдј йҖ’з»ҷ Page еҮҪж•°пјҢдҫҝеҸҜд»Ҙз”ҹжҲҗйқҷжҖҒйЎөйқўгҖӮеҚҡе®ўеҲ—иЎЁ URL жҳҜеӣәе®ҡзҡ„пјҢйӮЈд№ҲдёҚжҳҜеӣәе®ҡ URL зҡ„йЎөйқўпјҢиҰҒз”ҹжҲҗйқҷжҖҒйЎөйқўжҖҺд№ҲеҠһе‘ўпјҹжҜ”еҰӮеҚҡе®ўиҜҰжғ…йЎөгҖӮ
js
// pages/blog/[id].tsx
export async function getStaticPaths() {
  const articles = await fetch('https://localhost:3000/api/articles').then((res)=>res.json());
  return {
    paths: articles.map((p) => ({
      params: {
        id: p.id.toString(),
      },
    })),
    fallback: false,
  }
}

export const getStaticProps: GetStaticProps = async ({params}) => {
  const res = await fetch(`https://localhost:3000/api/articles/${params.id}`).then((res)=>res.json());

  return {
    props: { data: res },
  };
};

export default function Page({ data }: PageProps) {
  return (
    <div>
      <h1>{data.tltle}</h1>

    </div>

}

жҲ‘д»¬еҸҜд»ҘдҪҝз”Ё getStaticPaths иҺ·еҫ—жүҖжңүж–Үз« зҡ„и·Ҝеҫ„пјҢиҝ”еӣһзҡ„paths еҸӮж•°дјҡдј йҖ’з»ҷgetStaticPropsпјҢеңЁ getStaticPropsдёӯпјҢйҖҡиҝҮ params иҺ·еҫ—ж–Үз«  idпјҢ Next.js дјҡеңЁжһ„е»әж—¶пјҢе°Ҷpaths йҒҚеҺҶз”ҹжҲҗжүҖжңүйқҷжҖҒйЎөйқўгҖӮ
SSG зҡ„дјҳзӮ№е°ұжҳҜеҝ«пјҢйғЁзҪІдёҚйңҖиҰҒжңҚеҠЎеҷЁпјҢд»»дҪ•йқҷжҖҒжңҚеҠЎз©әй—ҙйғҪеҸҜд»ҘйғЁзҪІпјҢиҖҢзјәзӮ№д№ҹжҳҜеӣ дёәйқҷжҖҒпјҢдёҚиғҪеҠЁжҖҒжёІжҹ“пјҢжҜҸж·»еҠ дёҖзҜҮеҚҡе®ўпјҢе°ұйңҖиҰҒйҮҚж–°жһ„е»әгҖӮ

## ISRпјҲIncremental Static Regenerationпјү

дәҺжҳҜжңүдәҶдёҖеҸҰдёҖз§Қж–№жЎҲ ISRпјҢеўһйҮҸйқҷжҖҒз”ҹжҲҗпјҢеңЁи®ҝй—®ж—¶з”ҹжҲҗйқҷжҖҒйЎөйқўпјҢеңЁ Next.js дёӯпјҢе®ғжҜ” SSG ж–№жЎҲеҸӘйңҖиҰҒеҠ дәҶдёҖдёӘеҸӮж•°revalidate
js
export const getStaticProps: GetStaticProps = async () => {
  const res = await fetch('https://localhost:3000/api/articles').then((res)=>res.json());

  return {
    props: { data: res },
    revalidate: 20,
  };
};

дёҠйқўд»Јз ҒиЎЁзӨәпјҢеҪ“и®ҝй—®йЎөйқўж—¶пјҢеҸ‘зҺ° 20s жІЎжңүжӣҙж–°йЎөйқўе°ұдјҡйҮҚж–°з”ҹжҲҗж–°зҡ„йЎөйқўпјҢдҪҶеҪ“еүҚи®ҝй—®зҡ„иҝҳжҳҜе·Із»Ҹз”ҹжҲҗзҡ„йқҷжҖҒйЎөйқўпјҢд№ҹе°ұжҳҜпјҡжҳҜеҗҰйҮҚж–°з”ҹжҲҗйЎөйқўпјҢйңҖиҰҒж №жҚ®дёҠдёҖж¬Ўзҡ„з”ҹжҲҗж—¶й—ҙжқҘеҲӨж–ӯпјҢе№¶дё”ж•°жҚ®дјҡе»¶иҝҹ 1 ж¬ЎгҖӮ
жҲ‘д»¬еҸҜд»ҘеңЁйЎөйқўдёҠжҳҫзӨәз”ҹжҲҗж—¶й—ҙ

fallback жңү 3 дёӘеҖј

fallback: 'blocking' жңӘз”ҹжҲҗзҡ„йЎөйқўдҪҝз”ЁжңҚеҠЎз«ҜжёІжҹ“;
fallback: false жңӘз”ҹжҲҗзҡ„йЎөйқўи®ҝй—® 404
fallback: true еҪ“и®ҝй—®зҡ„йқҷжҖҒйЎөйқўдёҚеӯҳеңЁж—¶пјҢдјҡжҳҫзӨә loadingпјҢзӣҙеҲ°йқҷжҖҒйЎөйқўз”ҹжҲҗиҝ”еӣһж–°зҡ„йЎөйқўгҖӮ

жҲ‘д»¬е°Ҷ fallback и®ҫзҪ®дёә trueпјҢйҮҚж–°и®ҝй—®йЎөйқўгҖӮ

revalidateдјҡйўқеӨ–еҜјиҮҙжңҚеҠЎеҷЁжҖ§иғҪејҖй”ҖпјҢ20s з”ҹжҲҗдёҖж¬ЎйЎөйқўжҳҜжІЎеҝ…иҰҒзҡ„пјҢжҜ”еҰӮдёҖдәӣеҚҡе®ўзҪ‘з«ҷе’Ңж–°й—»зҪ‘з«ҷпјҢж–Үз« иҜҰжғ…еҸҳжӣҙжІЎйӮЈд№Ҳйў‘з№ҒгҖӮ

#### On-demand RevalidationпјҲжҢүйңҖеўһйҮҸз”ҹжҲҗпјү

иҮӘд»Һ next v12.2.0 ејҖе§Ӣж”ҜжҢҒжҢүйңҖеўһйҮҸз”ҹжҲҗпјҢжҲ‘д»¬еҸҜд»ҘеңЁ page зӣ®еҪ•дёӢж–°е»әдёҖдёӘ pages/api/revalidate.jsжҺҘеҸЈпјҢз”ЁдәҺи§ҰеҸ‘еўһйҮҸз”ҹжҲҗгҖӮ
js
// pages/api/revalidate.js

export default async function handler(req, res) {
  // и®ҫзҪ®дёҖдёӘз§ҳй’Ҙз”ЁдәҺжЈҖжҹҘпјҢи®ҝй—®еҗҲжі•жҖ§
  if (req.query.secret !== process.env.MY_SECRET_TOKEN) {
    return res.status(401).json({ message: 'Invalid token' })
  }

  try {
    // path дёәиҰҒи§ҰеҸ‘зҡ„е®һйҷ…и·Ҝеҫ„
    // e.g. for "/blog/[id]" this should be "/blog/5"
    await res.revalidate(req.query.path)
    return res.json({ revalidated: true })
  } catch (err) {
    return res.status(500).send('Error revalidating')
  }
}

жҜ”еҰӮжҲ‘д»¬еңЁж•°жҚ®еә“дёӯеўһеҠ дәҶ 2 жқЎж•°жҚ®пјҢжӯӨж—¶и®ҝй—® https://localhost:3000/api/revalidate?secret=<token>&path=/blog/5пјҢдҫҝеҸҜд»Ҙи§ҰеҸ‘пјҢз”ҹжҲҗж–°зҡ„йқҷжҖҒйЎөйқўдәҶгҖӮ

## Server component

Server component жҳҜ React18 жҸҗдҫӣзҡ„иғҪеҠӣпјҢ дёҺдёҠйқўзҡ„ SSR дёҚеҗҢпјҢзӣёеҪ“дәҺжҳҜжөҒејҸ SSRгҖӮ
дј з»ҹ SSR жү§иЎҢжӯҘйӘӨ

еңЁжңҚеҠЎеҷЁдёҠпјҢиҺ·еҸ–ж•ҙдёӘеә”з”Ёзҡ„ж•°жҚ®гҖӮ
еңЁжңҚеҠЎеҷЁдёҠпјҢе°Ҷж•ҙдёӘеә”з”ЁзЁӢеәҸж•°жҚ®жёІжҹ“дёә HTML е№¶еҸ‘йҖҒе“Қеә”гҖӮ
еңЁжөҸи§ҲеҷЁдёҠпјҢеҠ иҪҪж•ҙдёӘеә”з”ЁзЁӢеәҸзҡ„ JavaScript д»Јз ҒгҖӮ
еңЁе®ўжҲ·з«ҜпјҢе°Ҷ JavaScript йҖ»иҫ‘иҝһжҺҘеҲ°жңҚеҠЎз«Ҝиҝ”еӣһзҡ„ HTMLпјҲиҝҷе°ұжҳҜвҖңж°ҙеҗҲвҖқпјүгҖӮ

иҖҢд»ҘдёҠжҜҸдёӘжӯҘйӘӨеҝ…йЎ»е®ҢжҲҗпјҢжүҚеҸҜд»ҘејҖе§ӢдёӢдёҖдёӘжӯҘйӘӨгҖӮ

жҜ”еҰӮдёҖдёӘдј з»ҹзҡ„еҚҡе®ўйЎөйқўйҮҮз”Ё SSR зҡ„ж–№ејҸдҪҝз”Ё getServerSideProps зҡ„ж–№ејҸжёІжҹ“пјҢйӮЈд№Ҳе°ұйңҖиҰҒзӯү 3 дёӘжҺҘеҸЈе…ЁйғЁиҝ”еӣһжүҚеҸҜд»ҘзңӢеҲ°йЎөйқўгҖӮ

  const comments = await getComments()

еӣ жӯӨпјҢServer component и§ЈеҶідәҶ SSR дёӯзҡ„ 3 дёӘй—®йўҳ

> 1. дёҚеҝ…еңЁжңҚеҠЎеҷЁдёҠиҝ”еӣһжүҖжңүж•°жҚ®жүҚејҖе§Ӣиҝ”еӣһ htmlпјҢзӣёеҸҚжҲ‘д»¬еҸҜд»Ҙе…Ҳиҝ”еӣһдёҖдёӘ HTML з»“жһ„пјҢзӣёеҪ“дәҺйӘЁжһ¶еұҸгҖӮ
> 2. дёҚеҝ…зӯүеҫ…жүҖжңү JavaScript еҠ иҪҪе®ҢжҜ•жүҚиғҪејҖе§ӢиЎҘж°ҙгҖӮзӣёеҸҚпјҢжҲ‘д»¬еҸҜд»ҘеҲ©з”Ёд»Јз ҒжӢҶеҲҶдёҺжңҚеҠЎеҷЁжёІжҹ“з»“еҗҲдҪҝз”ЁпјҢReact е°ҶеңЁзӣёе…ід»Јз ҒеҠ иҪҪж—¶еҜ№е…¶иҝӣиЎҢж°ҙеҗҲгҖӮ
> 3. дёҚеҝ…зӯүеҫ…жүҖжңүз»„д»¶ж°ҙеҗҲе®ҢжҲҗпјҢйЎөйқўжүҚеҸҜд»ҘдәӨдә’гҖӮ

## жҖ»з»“

> CSR - е®ўжҲ·з«ҜжёІжҹ“гҖӮд№ҹе°ұжҳҜжҲ‘д»¬еёёиҜҙзҡ„ SPAпјҲsingle page applicationпјүпјҢдҪҝз”Ё useEffect иҺ·еҸ–жҺҘеҸЈж•°жҚ®пјҢдјҳзӮ№жҳҜеүҚз«ҜеҗҺз«Ҝе®Ңе…ЁеҲҶзҰ»пјҢйқҷжҖҒйғЁзҪІпјҢзјәзӮ№жҳҜ JavaScript иҝҮеӨ§пјҢдјҡйҖ жҲҗвҖңзҷҪеұҸвҖқпјҢзҪ‘йЎөеҲқе§Ӣ DOM дёәз©әпјҢдёҚеҲ©дәҺ SEOпјҢйҖӮеҗҲејҖеҸ‘дёҖдәӣеҗҺз«Ҝз®ЎзҗҶзі»з»ҹгҖӮ

> SSR - жңҚеҠЎеҷЁз«ҜжёІжҹ“гҖӮдјҳзӮ№жҳҜи§ЈеҶі SEO е’ҢзҷҪеұҸй—®йўҳпјҢзјәзӮ№жҳҜжҜҸж¬ЎжёІжҹ“йғҪдјҡиҜ·жұӮжңҚеҠЎеҷЁпјҢдјҡз»ҷжңҚеҠЎеҷЁйҖ жҲҗеҺӢеҠӣгҖӮ

> SSG - йқҷжҖҒз«ҷзӮ№з”ҹжҲҗгҖӮеңЁжһ„е»әж—¶иҺ·еҸ–ж•°жҚ®пјҢз”ҹжҲҗйқҷжҖҒйЎөйқўпјҢеҸӘйңҖиҰҒйқҷжҖҒйғЁзҪІпјҢйҖӮеҗҲејҖеҸ‘дёҖдәӣж•°жҚ®дёҚжҳ“еҸҳжӣҙзҡ„зҪ‘з«ҷпјҢжҜ”еҰӮејҖеҸ‘ж–ҮжЎЈгҖӮ

> ISR вҖ“ еўһйҮҸйқҷжҖҒеҶҚз”ҹгҖӮе®ғжҳҜ SSG е’Ң SSR зҡ„з»„еҗҲпјҢдё»иҰҒжҳҜйқ йқҷжҖҒжңҚеҠЎпјҢдҪҶеңЁж•°жҚ®иҝҮжңҹж—¶пјҢеҸҜд»ҘеҶҚж¬Ўд»Һ API иҺ·еҸ–ж•°жҚ®пјҢе№¶дё”з”ҹжҲҗйқҷжҖҒйЎөйқўпјҢжңҖйҖӮеҗҲеёёи§Ғзҡ„иө„и®Ҝзұ»гҖҒж–°й—»зұ»зҪ‘з«ҷгҖӮ

> Server component- д№ҹжҳҜ SSR зҡ„дёҖз§ҚпјҢ дҪҶдә’иЎҘдәҶ SSR зҡ„дёҚи¶іпјҢи®©зҪ‘йЎөжӢҘжңүжөҒејҸжёІжҹ“зҡ„иғҪеҠӣгҖӮ
