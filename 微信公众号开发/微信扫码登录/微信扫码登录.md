```js
    initWx() {
      const that = this;
      if (window.WxLogin) {
        that.initWxLogin();
      } else {
        var script = document.createElement('script');
        script.src =
          'https://res.wx.qq.com/connect/zh_CN/htmledition/js/wxLogin.js';
        script.onload = function () {
          that.initWxLogin();
        };
        script.onerror = function () {
          that.$Toast.error('网络问题，请刷新页面');
        };
        document.body.appendChild(script);
      }
    },
    initWxLogin() {
      this.step = '1';
      this.showQr = true;
      let vm = this;
      this.evloading = true;

      loginService
        .getInfo()
        .then((result) => {
          console.log(' loginService.getInfo', result);
          let info = result.data;

          let href =
            'data:text/css;base64,LmltcG93ZXJCb3ggLnFyY29kZSB7d2lkdGg6ODAlO21hcmdpbi10b3A6MH0uaW1wb3dlckJveCAudGl0bGUge2Rpc3BsYXk6bm9uZX0=';
          new WxLogin({
            self_redirect: true,
            id: 'wxLoginContainer',
            appid: info.appId,
            scope: info.scope,
            redirect_uri: encodeURIComponent(
                'https://你开放平台的绑定的域名/wechatlogin/' +
                process.env.VUE_APP_EV_DOMAIN +
                '/#/login/callback',
            ),
            state: info.state,
            href: href,
          });
        })
        .catch(function (error) {
          vm.evloading = false;
          console.log(error);
          vm.initWxLogin();
        });
    },
```
