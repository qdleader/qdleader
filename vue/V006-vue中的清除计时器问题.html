V006-vueдёӯзҡ„жё…йҷӨи®Ўж—¶еҷЁй—®йўҳвҳ…

1.жңүеҗҢеӯҰиҜҙжҲ‘йЎөйқўеҲҮжқҘеҲҮеҺ»дёәд»Җд№Ҳе°ұжҳҜжІЎжңүи§ҰеҸ‘destroyedе‘ўпјҢйӮЈжҲ‘зҡ„е®ҡж—¶еҷЁеІӮдёҚжҳҜйғҪжё…дёҚдәҶдәҶпјҢemmгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮ

destroyed() {
	if(timer) {
		clearInterval(timer);
	}
},

дёҚи§ҰеҸ‘жҳҜжӯЈеёёзҡ„е‘ҖгҖӮ

еӣ дёәbeforeDestoryе’ҢdestoryedжҳҜеңЁиҠӮзӮ№й”ҖжҜҒж—¶еҖҷжүҚдјҡи§ҰеҸ‘зҡ„пјҢеҰӮжһңдҪ еҲҮжҚўи·Ҝз”ұж—¶еҖҷжІЎжңүз”Ёv-ifжҺ§еҲ¶жҳҫйҡҗдёҚи§ҰеҸ‘д№ҹжҳҜжӯЈеёёзҡ„пјҢдёҖиҲ¬йӮЈе°ұдёҚдјҡи§ҰеҸ‘гҖӮзӯүдҪ зҰ»ејҖйЎөйқўж—¶еҖҷпјҢжүҚдјҡи§ҰеҸ‘гҖӮйӮЈд»–иҰҒдёҚзҰ»ејҖеҲҮдәҶдёӢи·Ҝз”ұиҝҳдёҖзӣҙиҜ·жұӮгҖӮгҖӮгҖӮ дёҚпјҒгҖӮпјҢиҝҷдёҚжҳҜжҲ‘жғіиҰҒзҡ„гҖӮ

йӮЈжҲ‘еҶҷдәҶдёӘи®Ўж—¶еҷЁиҝҳиҜ·дёҚдәҶдәҶпјҹ

жқҘжқҘжқҘпјҢе°‘е№ҙпјҢжҲ‘зңӢдҪ йӘЁйӘјжғҠеҘҮпјҢдј жҺҲдҪ жӯӨжі•гҖӮгҖӮ

еҲӨж–ӯеҪ“и·Ҝз”ұзҰ»ејҖж—¶еҖҷжё…йҷӨи®Ўж—¶еҷЁгҖӮ

    beforeRouteLeave(to, from, next){
         if(timer) {
            clearInterval(timer);
        }
        next();
    },

жӯӨжі•дёҚеҘҪдҪҝдҪ жқҘз ҚжҲ‘гҖӮгҖӮгҖӮ
