## 手机号加星号

```js
// 遮蔽手机号中间4位
const formatPhone = (row, column) => {
  const phone = row.phone
  if (!phone) return ''
  return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2')
}
```
